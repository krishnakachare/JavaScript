#### Getters and Setters
```javascript
let user = {
    firstName: 'John',
    lastName: 'Doe',
    get fullName() {
        return this.firstName + ' ' + this.lastName;
    },
    set fullName(name) {
        [this.firstName, this.lastName] = name.split(' ');
    }
};
user.fullName = 'Jane Smith';
console.log(user.fullName); // Output: Jane Smith
